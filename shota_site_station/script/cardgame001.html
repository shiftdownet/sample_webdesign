			endif

			gosub PaintPlayerSelectCard ; 選択中はずらして描写
			gosub PaintArrow            ; 矢印を描写
		endif
	else

		gosub CPU
		isEnd   = 1
	endif

;##########################################  初期化 init #############################################
; 変数･配列を初期化
*Initialization
	if state = 1 then
		msg 0
			初期化中…。
		endtext
	endif
	; getKey,timeKey初期化
	for _cnt0 = 0 to MAX_PLAYER
		for _cnt1 = 0 to MAX_CONFIG_KEY
			getKey[_cnt0][_cnt1] = 0
		endfor
	endfor
	; card初期化
	for _cnt0 = 0 to MAX_PLAYER
		for _cnt1 = 0 to MAX_CARD
			card[_cnt0][_cnt1] = -1
		endfor
	endfor
	; ary0初期化
	for _cnt0 = 0 to MAX_CARD
		ary0[_cnt0] = _cnt0
	endfor
	; card初期化
	for _cnt0 = 0 to MAX_CARD
		_var0 < _cnt0 - SUU_PLAYER * ( _cnt0 / SUU_PLAYER )
		_var1 < _cnt0 / SUU_PLAYER
		_max0 < MAX_CARD - _cnt0
		rand _max0 _rnd0
		card[_var0][_var1] = ary0[_rnd0]
		if ary0[_rnd0] = 1 then
			dia3 = _var0
		endif
		for _cnt1 = _rnd0 to _max0
			_var2 < _cnt1 + 1
			ary0[_cnt1] = ary0[_var2]
		endfor
	endfor
	; usedCard初期化
	for _cnt0 = 0 to MAX_CARD
		usedCard[_cnt0] = -1
	endfor
	; selectCard初期化
	for _cnt0 = 0 to SUU_SELECT
		selectCard[_cnt0] = -1
	endfor
	; pileCard初期化
	for _cnt0 = 0 to MAX_SELECT
		topCard[_cnt0] = -1
	endfor
	if funcState != 5 then
		; clearGame初期化
		for _cnt0 = 0 to MAX_PLAYER
			clearGame[_cnt0] = -1
		endfor
	endif
return

; 画像読み込み
*LoadGrapth
	if state = 1 then
		msg 0
			画像読み込み中…。
		endtext
	endif
	; カードフェイス読み込み準備
	for _cnt0 = 0 to MAX_NUMB
		for _cnt1 = 0 to MAX_MARK
			fileName = GRAPH_PATH
			fileName + FILE_NAME[_cnt1]
			if _cnt0 < 11 then
				_var0 < _cnt0 + 3
			else
				_var0 < _cnt0 - 10
			endif
			fileName + _var0
			fileName + EXTENSION
			_var1 < _cnt0 * SUU_MARK + _cnt1
			_var2 < _var1 + LAYER_FACE
			readyLoad $fileName$ CARD[_var1] _var2
		endfor
	endfor
	; カードジョーカー読み込み準備
	for _cnt0 = 0 to MAX_JOKER
		fileName = GRAPH_PATH
		fileName + FILE_NAME[4]
		fileName + _cnt0
		fileName + EXTENSION
		_var0 < MAX_CARD - MAX_JOKER + _cnt0
		_var1 < _var0 + LAYER_FACE
		readyLoad $fileName$ CARD[_var0] _var1
	endfor
	; カードバック読み込み準備
	for _cnt0 = 0 to MAX_CARD
		fileName = GRAPH_PATH
		fileName + FILE_NAME[5]
		fileName + EXTENSION
		_var0 < _cnt0 + LAYER_BACK
		readyLoad $fileName$ BACK[_cnt0] _var0
	endfor
	; アロー読み込み準備
	for _cnt0 = 0 to MAX_ARROW
		fileName = GRAPH_PATH
		fileName + FILE_NAME[6]
		fileName + COLOR_CODE[_cnt0]
		fileName + EXTENSION
		_var0 < _cnt0 + LAYER_ARROW
		readyLoad $fileName$ ARROW[_cnt0] _var0
	endfor
	; 順番を示す矢印
	arwGrp = _cnt0
	fileName = GRAPH_PATH
	fileName + FILE_NAME[7]
	fileName + EXTENSION
	_var0 < _cnt0 + LAYER_ARROW
	readyLoad $fileName$ ARROW[_cnt0] _var0
	for _cnt0 = 0 to 1
		fileName = GRAPH_PATH
		fileName + FILE_NAME[8]
		fileName + _cnt0
		fileName + PNG
		_var0 < _cnt0 + LAYER_REVO
		readyLoad $fileName$ REVO[_cnt0] _var0
	endfor
	for _cnt0 = 0 to 1
		fileName = GRAPH_PATH
		fileName + BACK_GRP[_cnt0]
		layer0 < LAYER_BGROUND + _cnt0
		readyLoad $fileName$ BACKGRP[_cnt0] layer0
	endfor
	; ロード開始
	startLoad

;##########################################  キー key #############################################
*CheckHitKey
	if player != player then
		player = 0 ; 未定義なら0
	endif
	isDown = 0
	for _cnt0 = 0 to player
		for _cnt1 = 0 to MAX_KEY
			_var0 < _cnt0 * ( MAX_KEY + 1 ) + _cnt1
			keydown configKey[_var0] _var1
			if _var1 = 1 then
				getKey[_cnt0][_cnt1] + 1
				isDown = 1
			else
				getKey[_cnt0][_cnt1] = 0
			endif
		endfor
	endfor
return

*PushAnyKey
	key KEY
	if KEY != 0 then
		return
	endif
	wait 1
goto PushAnyKey

*PushAnyKeyOrTime
	key KEY
	if KEY != 0 then
		return
	else if arg0 <= 0 then
		return
	endif
	arg0 - 1
	wait 1
goto PushAnyKeyOrTime

;##########################################  描写 paint #############################################
; プレイヤーのカードを描写
*PaintPlayerCard
	if CARD_CX[player] = 0 then
		mx = 1
		my = 0
	else
		mx = 0
		my = 1
	endif
	for _cnt0 = 0 to maxCard
		_var0 = card[player][_cnt0]
		_posX < CARD_CX[player] + mx * ( _cnt0 - maxCard / 2 ) * CARD_SPACE
		_posY < CARD_CY[player] + my * ( _cnt0 - maxCard / 2 ) * CARD_SPACE * -1
		if debug[0] = 0 then
			zoom         CARD[_var0] CARD_ZOOM CARD_ZOOM
			zoom         BACK[_var0] CARD_ZOOM CARD_ZOOM
			if playerAI[player] = 0 then
				locateCenter CARD[_var0] _posX _posY
				rotate       CARD[_var0] CARD_ROT[player]
				paint        CARD[_var0]
			else
				locateCenter BACK[_var0] _posX _posY
				rotate       BACK[_var0] CARD_ROT[player]
				paint        BACK[_var0]
			endif
		else
			locateCenter CARD[_var0] _posX _posY
			rotate       CARD[_var0] CARD_ROT[player]
			paint        CARD[_var0]
		endif
	endfor
return
; プレイヤーが選択しているカードをずらして描写
*PaintPlayerSelectCard
	gosub SuuSelectCard
	for _cnt0 = 0 to maxSelect
		_var0 = selectCard[_cnt0]
		_var0 = card[player][_var0]
		getGrpPos    CARD[_var0] _posX1 _posY1
		_posX2 = _posX1
		_posY2 < _posY1 - 10
		locateCenter CARD[_var0] _posX2 _posY2
		paint        CARD[_var0]
	endfor
return
; パイルのトップカードを描写
*PaintPileCard
	gosub SuuTopCard
	for _cnt0 = 0 to suuTop
		;何故か一部の変数名ではlocateCenterが動作しない
		;_var0 = topCard[_cnt0]
		;_posX < ( _cnt0 - suuTop / 2 ) * CARD_SPACE
		;_posY = 0
		;locateCenter CARD[_var0] _posX _posY
		;paint CARD[_var0]
		var0 = topCard[_cnt0]
		posX < ( _cnt0 - suuTop / 2 ) * CARD_SPACE
		posY = 0
		locateCenter CARD[var0] posX posY
		rotate       CARD[var0] 0
		paint        CARD[var0]
	endfor
return

; 矢印を描写
*PaintArrow
	gosub EraseArrow
	if suuSelect = SUU_SELECT then
		arrow = 0
	else if suuSelect = 0 then
		arrow = 1
	else
		arrow = 2
	endif
	_posX < CARD_CX[player] + ( isSelect - maxCard / 2 ) * CARD_SPACE - ( CARD_SIZEX * CARD_ZOOM / 200 ) + 8
	_posY < CARD_CY[player] - 10 - ( CARD_SIZEY * CARD_ZOOM / 200 ) - 8
	;CARD_ZOOM
	locateCenter ARROW[arrow] _posX _posY
	paint        ARROW[arrow]
return
; 描写したカードを消去
*EraseCard
	if player != player then
		; playerが未定義なら全てのカードを消去
		player = 0
		_max0 = MAX_PLAYER
	else
		; playerが既定義ならそのプレイヤーのカードのみ消去
		_max0 = player
