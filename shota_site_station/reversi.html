; (x,y)において裏返すことのできる石の数を取得する。flipがTRUEなら実際に裏返す。
; 引数   : (x,y)     => 座標
; 〃     : flip      => フラグ(TRUEなら実際に裏返す)
; 戻り値 : flipCount => 裏返すことのできる石の数
*CheckCell
	flipCount = 0
	nextPlayer < 3 - player
	if board[x][y] = 0 then ; セルが空
		for dx = -1 to 1
			for dy = -1 to 1
				if dx = 0 then
					if dy = 0 then
						dy + 1 ; (dx,dy)=(0,0)ならdyを1進める
					endif
				endif
				nx < x + dx
				ny < y + dy
				var0 = 0
				for while = 0 to 0 step 0
					if board[nx][ny] = nextPlayer then
						nx + dx
						ny + dy
						var0 + 1
					else
						while = 1
					endif
				endfor
				if var0 > 0 then
					if board[nx][ny] = player then
						flipCount + var0
						if flip = TRUE then
							nx < x + dx
							ny < y + dy
							for while = 0 to 0 step 0
								if board[nx][ny] = nextPlayer then
									board[nx][ny] = player
									nx + dx
									ny + dy
								else
									while = 1
								endif
							endfor
							board[x][y] = player
						endif
					endif
				endif
			endfor
		endfor
	endif
return
; 全ての座標でおけるかどうかチェックする
; 引数   : arg0 => 0=ノーマル 1=石を一番とれる 2=石を一番とれない
; 〃     : arg1 => 0=評価する 1=評価しない
; 戻り値 :

	suu   = 0 ; 
	point = 100 ; 得点
	if arg0 = 2 then
		suu = 100
	endif
	if arg1 = 1 then
		point = -100
	endif
	rand 3 rnd

				var0 = 0
				var1 = 0
			else if arg0 = 1 then
				var0 = suu
				var1 = flipCount
			else if arg0 = 2 then
				var0 = flipCount
				var1 = suu
			endif
			if flipCount > 0 then
				if var0 <= var1 then
					if BOARD_POINT[x][y] >= point then
						suu = flipCount
						_x  = x
						_y  = y
						point = BOARD_POINT[x][y]
					else
						if arg1 = 0 then
							suu = flipCount
							_x  = x
							_y  = y
						endif
					endif
				else if BOARD_POINT[x][y] >= point then
					suu   = flipCount
					_x  = x
					_y  = y

	endfor

	y = _y

*ClickCell
	mouse mx my mc
	gosub MoveMouse
	if ret0 = TRUE then
		var0 < BOARD_POSX + CELL_SIZE / 2
		var1 < BOARD_POSY + CELL_SIZE / 2
		if mx > var0 then
			if my > var1 then
				for cnt0 = BOARD_CELL_MIN to BOARD_CELL_MAX
					for cnt1 = BOARD_CELL_MIN to BOARD_CELL_MAX
						posX0 < cnt0 * CELL_SPACE + BOARD_POSX
						posY0 < cnt1 * CELL_SPACE + BOARD_POSY
						posX1 < posX0 + ( CELL_SIZE / 2 )
						posY1 < posY0 + ( CELL_SIZE / 2 )
						if mx < posX1 then
							if my < posY1 then
								locateCenter MOUSE posX0 posY0
								paint        MOUSE
								ret0 = 1
								x = cnt0
								y = cnt1
								return
							endif
						endif
					endfor
				endfor
			endif
		endif
		erase MOUSE
		ret0 = 0
	endif
return
; マウス操作においてなんらかのアクションが起こるとTRUEを返す
; 引数   : void
; 戻り値 : ret0
*MoveMouse
	if __mx1 != __mx1 then
		mouse __mx1 __my1 __mc1
	endif
	mouse __mx2 __my2 __mc2
	__var0 < ( __mx1 - __mx2 ) + ( __my1 - __my2 )
	if __var0 != 0 then
		__mx1 = undefined
		ret0  = TRUE
	else if __mc2 = 1 then
		ret0  = TRUE
	else
		ret0 = FALSE
	endif
return

		for cnt1 = BOARD_CELL_MIN to BOARD_CELL_MAX
			for cnt2 = BOARD_CELL_MIN to BOARD_CELL_MAX
				var0  < cnt0 * ( BOARD_ARY_MAX * BOARD_ARY_MAX + 1 ) + cnt1 * ( BOARD_ARY_MAX + 1 ) + cnt2
				layer < var0 + LAYER_CELL
				readyLoad $fileName$ CELL[var0] layer
			endfor
		endfor
